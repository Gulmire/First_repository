package package1;

//inheritance
public class Car extends Vehicle {
	// method overriding
	public void run() {
		System.out.println("cars run");
	}

	// method overloading
	public void add_numberOfCar(int a, int b) {
		int c = a + b;
		System.out.println(c);
	}

	public void add_numberOfCar(int a, int b, int c) {
		int d = a + b + c;
		System.out.println(d);
	}
}
